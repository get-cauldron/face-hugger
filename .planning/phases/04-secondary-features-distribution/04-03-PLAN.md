```rust
pub fn build_tray_menu(app: &AppHandle, active: usize, pct: u8) -> tauri::Result<Menu<tauri::Wry>> {
    let status_text = if active > 0 {
        format!("{} upload(s) — {}%", active, pct)
    } else {
        "No active uploads".to_string()
    };

    let menu = Menu::with_items(app, &[
        &MenuItem::with_id(app, "status", &status_text, false, None::<&str>)?,
        &PredefinedMenuItem::separator(app)?,
        &MenuItem::with_id(app, "open", "Open Face Hugger", true, None::<&str>)?,
        &MenuItem::with_id(app, "pause", "Pause All", active > 0, None::<&str>)?,
        &PredefinedMenuItem::separator(app)?,
        &MenuItem::with_id(app, "quit", "Quit", true, None::<&str>)?,
    ])?;

    Ok(menu)
}
```

**Setup tray function (called from setup hook):**
```rust
pub fn setup_tray(app: &AppHandle) -> tauri::Result<()> {
    let menu = build_tray_menu(app, 0, 0)?;

    let _tray = TrayIconBuilder::with_id(app, "main")
        .icon(app.default_window_icon().unwrap().clone())
        .menu(&menu)
        .menu_on_left_click(false)
        .on_menu_event(|app, event| {
            match event.id.as_ref() {
                "open" => {
                    if let Some(window) = app.get_webview_window("main") {
                        let _ = window.show();
                        let _ = window.set_focus();
                    }
                }
                "pause" => {
                    // Emit pause-all event to frontend (which calls pause_all_uploads command)
                    let _ = app.emit("tray-pause-all", ());
                }
                "quit" => {
                    app.exit(0);
                }
                _ => {}
            }
        })
        .on_tray_icon_event(|tray, event| {
            // Left-click reopens window (macOS/Windows only — Linux does not fire this)
            if let TrayIconEvent::Click {
                button: MouseButton::Left,
                button_state: MouseButtonState::Up,
                ..
            } = event {
                let app = tray.app_handle();
                if let Some(window) = app.get_webview_window("main") {
                    let _ = window.show();
                    let _ = window.set_focus();
                }
            }
        })
        .build()?;

    Ok(())
}
```

**Tray animation functions:**
```rust
pub fn start_tray_animation(app: AppHandle) {
    let state = app.state::<AppState>();
    // Cancel existing animation if any
    let mut anim = state.tray_animation.blocking_lock();
    if let Some(handle) = anim.take() {
        handle.abort();
    }

    let app_clone = app.clone();
    let handle = tokio::spawn(async move {
        // Load animation frames — use embedded bytes or file-based
        // For simplicity, create 4 opacity levels by alternating icons
        let default_icon = app_clone.default_window_icon().unwrap().clone();
        // Simple pulsing: alternate between default icon and slightly different variant
        // In practice, load from icons/tray-frame-*.png files
        let mut frame_idx = 0u8;
        loop {
            if let Some(tray) = app_clone.tray_by_id("main") {
                // Pulse effect: cycle through frames
                // For now, just use default icon (animation frames are a visual polish item)
                let _ = tray.set_icon(Some(default_icon.clone()));
            }
            frame_idx = frame_idx.wrapping_add(1);
            tokio::time::sleep(std::time::Duration::from_millis(150)).await;
        }
    });

    *anim = Some(handle);
}

pub fn stop_tray_animation(app: &AppHandle) {
    let state = app.state::<AppState>();
    let mut anim = state.tray_animation.blocking_lock();
    if let Some(handle) = anim.take() {
        handle.abort();
    }
    // Reset to static icon
    if let Some(tray) = app.tray_by_id("main") {
        if let Some(icon) = app.default_window_icon() {
            let _ = tray.set_icon(Some(icon.clone()));
        }
    }
}
```

```rust
    // ... all existing plugins, invoke_handler, setup ...
    .build(tauri::generate_context!())

app.run(|app_handle, event| {
    match event {
        tauri::RunEvent::ExitRequested { api, code, .. } => {
            // code == None means window close button clicked (not process kill)
            if code.is_none() {
                // Check if uploads are active

                    // Hide main window instead of closing

                        let _ = w.hide();
                    }
                }
                // If no active uploads, allow natural exit
            }
        }
        _ => {}
    }
});
```

```rust
fn check_active_uploads(app_handle: &AppHandle) -> bool {
    // Read from the upload progress map or DB to check active count
    let state = app_handle.state::<AppState>();
    let progress = state.progress_map.lock().unwrap();
    // Active uploads are those with bytes_transferred < bytes_total and status "uploading"
```
