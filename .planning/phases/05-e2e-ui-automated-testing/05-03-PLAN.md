    test "token paste login succeeds":

    2. Click "Use access token instead" to switch to token mode

    2. page.evaluate() to set validate_token handler to throw 'Invalid token'
    3. Fill input, click submit
    4. Assert error text "Invalid token" is visible

    test "OAuth mode shows waiting state":
    1. App loads in OAuth mode
    2. page.evaluate() to set oauth_start handler to return 'https://huggingface.co/oauth/authorize'

    4. Click "Sign in with Hugging Face"
    5. Assert "Waiting for browser login..." text appears
    6. Assert "Cancel Sign-in" button is visible

    test "logout clears session":
    1. page.evaluate() to set check_existing_token to return { token: 'hf_test', user: mockUserInfo } so app starts authenticated
    2. page.goto('/') to reload with authenticated state
    3. Find and click logout button/link in sidebar
    4. Assert login screen reappears
