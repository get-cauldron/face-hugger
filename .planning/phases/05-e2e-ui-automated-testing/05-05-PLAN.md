    2. Create `tests/native/wdio.conf.js` following RESEARCH.md Pattern 4:
       ```javascript
       const { spawn } = require('child_process');
       const path = require('path');

       // Determine binary path based on platform
       const isWindows = process.platform === 'win32';
       const binaryName = isWindows ? 'face-hugger.exe' : 'face-hugger';
       const binaryPath = path.resolve(__dirname, '../../src-tauri/target/release', binaryName);

       exports.config = {
         specs: ['./specs/**/*.spec.js'],
         maxInstances: 1,
         capabilities: [{
           maxInstances: 1,
           'tauri:options': {
             application: binaryPath,
           },
           browserName: 'wry',
         }],
         framework: 'mocha',
         reporters: ['spec'],
         mochaOpts: {
           timeout: 60000,
         },
         onPrepare: function () {
           // Start tauri-driver before tests
           this.tauriDriver = spawn('tauri-driver', [], {
             stdio: [null, process.stdout, process.stderr],
           });
         },
         onComplete: function () {
           if (this.tauriDriver) {
             this.tauriDriver.kill();
           }
         },
       };
       ```

    3. Create `tests/native/specs/tray.spec.js`:
       - Test "app has system tray icon": After app launches via WebdriverIO, verify the app process is running (the tray icon is created during app setup in lib.rs). Since WebDriver cannot directly query OS-level tray icons, verify the app launched successfully and the main window is accessible.
       - Test "app window exists and is accessible": Use browser.getTitle() or browser.getUrl() to verify the WebView is responding. Assert the title or URL is not empty.
       - Note: Direct tray menu interaction via WebDriver is limited. These tests validate the app binary launches correctly with tray support enabled. More granular tray testing would require platform-specific automation tools.

    4. Create `tests/native/specs/window.spec.js`:
       - Test "app launches and shows content": Assert browser.getTitle() returns a non-empty string
       - Test "app window has expected dimensions": Use browser.getWindowSize() to check width and height are reasonable (e.g., width > 400, height > 300)
       - Test "app renders login screen on fresh launch": Use browser.$() to find the login screen element, assert it exists (the app starts unauthenticated with no stored token)

    Note: These tests require a compiled Tauri binary (`npm run tauri build -- --debug`). They will NOT run on macOS due to the lack of a WKWebView WebDriver (RESEARCH.md Pitfall 2).

    Add the following steps to the test job:

    ```yaml
    # tauri-driver native tests (Linux/Windows only -- no WKWebView driver on macOS)
    - name: Install tauri-driver
      if: matrix.platform != 'macos-latest'
      run: cargo install tauri-driver --locked

    - name: Build Tauri binary for native tests
      if: matrix.platform != 'macos-latest'
      run: npx tauri build --debug
      env:
        TAURI_SIGNING_PRIVATE_KEY: ${{ secrets.TAURI_SIGNING_PRIVATE_KEY }}
        TAURI_SIGNING_PRIVATE_KEY_PASSWORD: ${{ secrets.TAURI_SIGNING_PRIVATE_KEY_PASSWORD }}

    - name: Install native test dependencies
      if: matrix.platform != 'macos-latest'
      run: npm ci --prefix tests/native

    - name: Install webkit2gtk-driver (Ubuntu)
      if: matrix.platform == 'ubuntu-22.04'
      run: sudo apt-get install -y webkit2gtk-driver

    - name: Native tests (Linux - headless via xvfb)
      if: matrix.platform == 'ubuntu-22.04'
      run: xvfb-run npm run test:native --prefix tests/native

    - name: Native tests (Windows)
      if: matrix.platform == 'windows-latest'
      run: npm run test:native --prefix tests/native
    ```

    Important: The `if: matrix.platform != 'macos-latest'` condition ensures no tauri-driver tests run on macOS. Ubuntu needs `xvfb-run` for headless WebDriver execution. The binary build step uses `--debug` for faster compilation (release optimizations not needed for testing).
