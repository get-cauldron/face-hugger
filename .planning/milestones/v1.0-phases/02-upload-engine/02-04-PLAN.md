```rust
use std::time::Duration;

/// Exponential backoff: 5s, 15s, 30s, 60s cap — with +/-20% jitter.
/// Per CONTEXT.md locked decision.
pub fn backoff_duration(retry_count: u32) -> Duration {
    let base_ms: u64 = match retry_count {
        0 => 5_000,
        1 => 15_000,
        2 => 30_000,
        _ => 60_000,
    };
    let jitter = (rand::random::<f64>() * 0.4 - 0.2) * base_ms as f64;
    Duration::from_millis((base_ms as f64 + jitter).max(1000.0) as u64)
}
```

2. Create `src-tauri/src/upload/worker.rs` with the main upload function:

**run_upload_job** — The main entry point for executing an upload job:
```rust
pub async fn run_upload_job(
    job_id: String,
    db: SqlitePool,
    client: reqwest::Client,
    hf_token: String,
    cancel_token: CancellationToken,
    progress_state: Arc<std::sync::Mutex<HashMap<String, UploadProgress>>>,
) -> Result<(), String>
```

The function follows this sequence:

**Step 1: Load job from DB**
- `get_job(&db, &job_id)` — if None, return error

**Step 2: Verify source file exists**
- Check `std::path::Path::new(&job.file_path).exists()`
- If not: update job state to "failed" with error "Source file changed or missing — please re-add the file", return error

**Step 3: Check write access**
- `check_write_access(&client, &hf_token, &job.repo_type, &job.repo_id).await`
- On error: update job to "failed" with the error message (permission denied = not retryable)

**Step 4: Hash file (state -> "hashing")**
- Update job state to "hashing" in DB
- Call `hash_file_streaming(Path::new(&job.file_path)).await`
- Wrap in `tokio::select!` with cancel_token:
  ```rust
  tokio::select! {
      _ = cancel_token.cancelled() => { /* check if paused or cancelled in DB, return accordingly */ }
      result = hash_file_streaming(&path) => { /* continue with hash */ }
  }
  ```
- Store SHA-256 hash bytes and verify total_bytes matches job.total_bytes (update if 0)
- Convert hash to hex string: `let file_oid = hex::encode(&hash_bytes);`

**Step 5: Detect protocol (if not already set)**
- If `job.protocol` is None: call `detect_upload_protocol(&client, &hf_token, &job.repo_id, &file_oid, total_bytes).await`
- Update protocol in DB via `update_job_protocol`
- If already set (resume case): use stored protocol

**Step 6: Branch on protocol**

**Xet path:**
- Update state to "uploading"
- Get Xet write token: `get_xet_write_token(&client, &hf_token, &job.repo_type, &job.repo_id, &job.revision).await`
- Get already confirmed xorbs from DB: `get_confirmed_xorbs(&db, &job_id).await`
- **Read file in streaming chunks** — use a simplified chunking approach for now:
  - Read file in fixed 64MB blocks (this is a simplification; content-defined chunking with Gearhash is ideal but can be added later)
  - Each 64MB block = one xorb
  - Compute the SHA-256 of each block, encode with `xet_hash_to_hex`
  - If xorb hash is in confirmed set: skip (resume)
  - Otherwise: upload via `upload_xorb(&client, &token.cas_url, &token.access_token, &xorb_hash, xorb_bytes).await`
  - On success: `insert_confirmed_xorb(&db, &job_id, &xorb_hash).await`
  - Update bytes_confirmed in progress_state
  - Before each xorb: check `token_needs_refresh(&xet_token)` — if true, refresh token
  - Wrap each xorb upload in `tokio::select!` with cancel_token
  - On network error: apply backoff, increment retry_count, retry the xorb (not the whole file)
- **After ALL xorbs uploaded:** Build and upload shard
  - Shard contains: list of (xorb_hash, chunk_byte_ranges) mapping file offsets to xorb contents
  - For the simplified fixed-size chunking: shard is a serialized list of xorb hashes with their byte ranges
  - NOTE: The actual xorb and shard binary formats require reading the HF spec pages (`/docs/xet/en/xorb`, `/docs/xet/en/shard`). For now, implement a minimal working version that:
    - Serializes xorb as raw bytes (the concatenated chunk data)
    - Serializes shard as a JSON or msgpack list of {xorb_hash, offset, length} (placeholder — will need spec-compliant format)
    - Mark with TODO comments for spec-compliant serialization
  - `upload_shard(&client, &token.cas_url, &token.access_token, shard_bytes).await`

**LFS multipart path:**
- Update state to "uploading"
- Get upload info: `get_lfs_upload_info(&client, &hf_token, &job.repo_id, &file_oid, total_bytes).await`
- Get confirmed parts: `get_confirmed_lfs_parts(&db, &job_id).await`
- For each part in upload_info.parts:
  - If part_number in confirmed set: skip (resume)
  - Read the corresponding byte range from the file
  - PUT to the pre-signed URL with the part bytes
  - On success: `insert_confirmed_lfs_part(&db, &job_id, part.part_number).await`
  - Update bytes_confirmed in progress_state
  - Wrap each part upload in `tokio::select!` with cancel_token
  - On 403 from S3: URLs expired — re-request fresh URLs via `get_lfs_upload_info`, retry
  - On network error: apply backoff, retry the part

**Step 7: Commit (state -> "committing")**
- `create_commit(&client, &hf_token, &job.repo_type, &job.repo_id, &job.revision, &job.commit_message, &job.file_name, &file_oid, total_bytes).await`
- On success: update state to "done"
- On error: update state to "failed" with error

**Step 8: Cleanup**
- Remove cancel token from registry
- Update final progress state

**Error handling throughout:**
- Network errors: log, apply backoff, retry (unlimited retries per CONTEXT.md)
- 401 from CAS: refresh Xet token and retry
- 403 from S3 (LFS): re-request pre-signed URLs and retry
- Source file changed: stat the file before each chunk read — if mtime changed or file missing, fail with "Source file changed or missing"
- All retryable errors: `backoff_duration(retry_count)` then retry
- Non-retryable errors (permission denied, bad request): fail immediately

```rust
mod tests {
    use super::*;

    #[test]
    fn test_backoff_schedule() {
        // Test multiple times due to jitter
        for _ in 0..10 {
            let d0 = backoff_duration(0);
            assert!(d0.as_millis() >= 4000 && d0.as_millis() <= 6000, "retry 0 ~5s: got {}ms", d0.as_millis());
            let d1 = backoff_duration(1);
            assert!(d1.as_millis() >= 12000 && d1.as_millis() <= 18000, "retry 1 ~15s: got {}ms", d1.as_millis());
            let d2 = backoff_duration(2);
            assert!(d2.as_millis() >= 24000 && d2.as_millis() <= 36000, "retry 2 ~30s: got {}ms", d2.as_millis());
            let d3 = backoff_duration(3);
            assert!(d3.as_millis() >= 48000 && d3.as_millis() <= 72000, "retry 3+ ~60s: got {}ms", d3.as_millis());
        }
    }

    #[test]
    fn test_backoff_caps_at_60s() {
        for retry in 3..100 {
            let d = backoff_duration(retry);
            assert!(d.as_millis() <= 72000, "Backoff must cap near 60s");
        }
    }
}
```

```rust
        // For a 150MB file with 64MB xorb size:
        let total = 150 * 1024 * 1024u64;
        let xorb_size = 64 * 1024 * 1024u64;
        let ranges = compute_xorb_ranges(total, xorb_size);
```
