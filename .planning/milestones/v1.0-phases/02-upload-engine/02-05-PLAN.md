**SpeedTracker** — Sliding window speed calculation:
```rust
pub struct SpeedTracker {
    samples: Vec<(std::time::Instant, u64)>, // (timestamp, bytes_sent)
    window: std::time::Duration,
}

impl SpeedTracker {
    pub fn new() -> Self {
        Self {
            samples: Vec::new(),
            window: std::time::Duration::from_secs(5), // 5-second sliding window
        }
    }

    pub fn record(&mut self, bytes_sent: u64) {
        let now = std::time::Instant::now();
        self.samples.push((now, bytes_sent));
        // Prune old samples
        self.samples.retain(|(t, _)| now.duration_since(*t) < self.window);
    }

    pub fn speed_bps(&self) -> f64 {
        if self.samples.len() < 2 { return 0.0; }
        let first = self.samples.first().unwrap();
        let last = self.samples.last().unwrap();
        let elapsed = last.0.duration_since(first.0).as_secs_f64();
        if elapsed < 0.001 { return 0.0; }
        let bytes_delta = last.1.saturating_sub(first.1);
        bytes_delta as f64 / elapsed
    }

    pub fn eta_seconds(&self, total_bytes: u64, bytes_sent: u64) -> f64 {
        let speed = self.speed_bps();
        if speed < 1.0 { return f64::INFINITY; }
        let remaining = total_bytes.saturating_sub(bytes_sent) as f64;
        remaining / speed
    }
}
```

**start_progress_emitter** — Spawns a timer task that reads ProgressMap and sends via Channel:
```rust
use tauri::ipc::Channel;
use tokio::time::{interval, Duration};

pub fn start_progress_emitter(
    progress_map: ProgressMap,
    channel: Channel<Vec<UploadProgress>>,
) -> tokio::task::JoinHandle<()> {
    tokio::spawn(async move {
        let mut timer = interval(Duration::from_millis(500));
        loop {
            timer.tick().await;
            let updates: Vec<UploadProgress> = {
                let map = progress_map.lock().unwrap();
                map.values().cloned().collect()
            };
            if !updates.is_empty() {
                let _ = channel.send(updates);
            }
        }
    })
}
```
The channel sends a `Vec<UploadProgress>` (all active jobs) every 500ms. This is more efficient than one channel per job.

**set_concurrent_limit** — Update the concurrent upload limit (user-configurable, stored in tauri-plugin-store):
```rust
#[tauri::command]
#[specta::specta]
pub async fn set_concurrent_limit(
    limit: usize,
    app: tauri::AppHandle,
    state: tauri::State<'_, AppState>,
) -> Result<(), String> {
    // Clamp to 1..=5 range (mirrors UploadQueue::set_max_concurrent clamping)
    let clamped = limit.clamp(1, 5);

    // Persist to tauri-plugin-store so value survives restart
    let store = app.store("preferences.json").map_err(|e| e.to_string())?;
    store.set("concurrent_upload_limit", serde_json::json!(clamped));
    store.save().map_err(|e| e.to_string())?;

    // Apply immediately to the running queue
    let mut queue = state.upload_queue.lock().await;
    queue.set_max_concurrent(clamped);

    Ok(())
}
```
Clamping is intentional defense-in-depth — both the command and `set_max_concurrent` clamp, so callers cannot set out-of-range values even through direct Rust calls.

  id: string;
  file_path: string;
  file_name: string;
  repo_id: string;
  repo_type: string;
  revision: string;
  commit_message: string;
  total_bytes: number;
  bytes_confirmed: number;
  protocol: 'xet' | 'lfs_multipart' | null;
  state: 'pending' | 'hashing' | 'uploading' | 'committing' | 'done' | 'failed' | 'paused' | 'cancelled';
  priority: boolean;
  retry_count: number;
  last_error: string | null;
  created_at: number;
  updated_at: number;
}

export interface UploadProgress {
  job_id: string;
  bytes_sent: number;
  total_bytes: number;
  speed_bps: number;
  eta_seconds: number;
  state: UploadJob['state'];
}

// Helper to unwrap tauri-specta Result<T, E>
function unwrap<T>(result: { status: 'ok'; data: T } | { status: 'error'; error: string }): T {
  if (result.status === 'error') throw new Error(result.error);
  return result.data;
}

export async function enqueueUpload(params: {
  filePath: string;
  repoId: string;
  repoType: string;
  revision?: string;
  commitMessage: string;
  priority?: boolean;
}): Promise<UploadJob> {
  const result = await invoke('enqueue_upload', {
    filePath: params.filePath,
    repoId: params.repoId,
    repoType: params.repoType,
    revision: params.revision ?? 'main',
    commitMessage: params.commitMessage,
    priority: params.priority ?? false,
  });
  return unwrap(result as any);
}

export async function cancelUpload(jobId: string): Promise<void> {
  const result = await invoke('cancel_upload', { jobId });
  unwrap(result as any);
}

export async function pauseUpload(jobId: string): Promise<void> {
  const result = await invoke('pause_upload', { jobId });
  unwrap(result as any);
}

export async function resumeUpload(jobId: string): Promise<void> {
  const result = await invoke('resume_upload', { jobId });
  unwrap(result as any);
}

export async function pauseAllUploads(): Promise<number> {
  const result = await invoke('pause_all_uploads');
  return unwrap(result as any);
}

export async function listUploads(): Promise<UploadJob[]> {
  const result = await invoke('list_uploads');
  return unwrap(result as any);
}

export async function setUploadPriority(jobId: string, priority: boolean): Promise<void> {
  const result = await invoke('set_upload_priority', { jobId, priority });
  unwrap(result as any);
}

export function startUploadMonitoring(
  onProgress: (updates: UploadProgress[]) => void
): Promise<void> {
  const channel = new Channel<UploadProgress[]>();
  channel.onmessage = onProgress;
  return invoke('start_upload_monitoring', { channel }) as Promise<void>;
}

/** Update concurrent upload limit (1-5). Persisted to tauri-plugin-store. */
export async function setConcurrentLimit(limit: number): Promise<void> {
  const result = await invoke('set_concurrent_limit', { limit });
  unwrap(result as any);
}
