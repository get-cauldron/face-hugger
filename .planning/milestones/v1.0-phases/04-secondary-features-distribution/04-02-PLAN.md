```typescript
/** Feature (column) metadata from /rows and /search endpoints */
export interface DatasetFeature {
  feature_idx: number;
  name: string;
  type: { dtype?: string; _type: string; names?: string[] };
}

/** Single row from /rows, /search, /filter */
export interface DatasetRow {
  row_idx: number;
  row: Record<string, unknown>;
  truncated_cells: string[];
}

/** Response from GET /rows */
export interface DatasetRowsResponse {
  features: DatasetFeature[];
  rows: DatasetRow[];
  num_rows_total: number;
  num_rows_per_page: number;
  partial: boolean;
}

/** Response from GET /search (same structure as /rows + search-specific) */
export interface DatasetSearchResponse extends DatasetRowsResponse {}

/** Response from GET /filter (same structure as /rows) */
export interface DatasetFilterResponse extends DatasetRowsResponse {}

/** Response from GET /splits */
export interface DatasetSplitsResponse {
  splits: Array<{
    dataset: string;
    config: string;
    split: string;
  }>;
}

/** Response from GET /is-valid */
export interface DatasetIsValidResponse {
  viewer: boolean;
  preview: boolean;
  search: boolean;
  filter: boolean;
  statistics: boolean;
}

/** Per-column statistics from GET /statistics */
export interface ColumnStatistics {
  column_name: string;
  column_type: 'int' | 'float' | 'bool' | 'class_label' | 'string_label' | 'string_text' | 'list' | 'audio' | 'image' | 'datetime';
  column_statistics: {
    nan_count: number;
    nan_proportion: number;
    min?: number | string;
    max?: number | string;
    mean?: number | string;
    median?: number | string;
    std?: number | string;
    n_unique?: number;
    frequencies?: Record<string, number>;
    histogram?: { hist: number[]; bin_edges: (number | string)[] };
  };
}

/** Response from GET /statistics */
export interface DatasetStatisticsResponse {
  num_examples: number;
  statistics: ColumnStatistics[];
  partial: boolean;
}
```

**4. Create src/queries/useDatasetViewer.ts with TanStack Query hooks:**

Base URL constant:

Helper to get token:
```typescript
import { useAuthStore } from '@/stores/authStore';

function authHeaders(): HeadersInit {
  const token = useAuthStore.getState().token;
  return token ? { Authorization: `Bearer ${token}` } : {};
}
```

Hooks (all use `useQuery` from `@tanstack/react-query`):

- `useDatasetIsValid(dataset: string)` — GET `/is-valid?dataset={dataset}`, returns `DatasetIsValidResponse`. `enabled: !!dataset`. staleTime: 10 minutes.

- `useDatasetSplits(dataset: string, enabled: boolean)` — GET `/splits?dataset={dataset}`, returns `DatasetSplitsResponse`. `enabled` param gates the query (only fetch after is-valid check). staleTime: 10 minutes.

- `useDatasetRows(dataset: string, config: string, split: string, page: number, enabled: boolean)` — GET `/rows?dataset={dataset}&config={config}&split={split}&offset={page * PAGE_SIZE}&length={PAGE_SIZE}`, returns `DatasetRowsResponse`. staleTime: 5 minutes. `keepPreviousData: true` for smooth pagination.

- `useDatasetSearch(dataset: string, config: string, split: string, query: string, page: number, enabled: boolean)` — GET `/search?dataset={dataset}&config={config}&split={split}&query={query}&offset={page * PAGE_SIZE}&length={PAGE_SIZE}`, returns `DatasetSearchResponse`. `enabled` gated on `query.length > 0 && enabled`. staleTime: 2 minutes.

- `useDatasetFilter(dataset: string, config: string, split: string, where_clause: string, page: number, enabled: boolean)` — GET `/filter?dataset={dataset}&config={config}&split={split}&where={encodeURIComponent(where_clause)}&offset={page * PAGE_SIZE}&length=${PAGE_SIZE}`, returns `DatasetFilterResponse`. `enabled` gated on `where_clause.length > 0 && enabled`. staleTime: 2 minutes.

- `useDatasetStatistics(dataset: string, config: string, split: string, enabled: boolean)` — GET `/statistics?dataset={dataset}&config={config}&split={split}`, returns `DatasetStatisticsResponse`. staleTime: 10 minutes.

Each hook follows the pattern from `useRepoFiles.ts`: uses `useQuery`, includes error handling, and returns the standard TanStack Query result object. All URL parameters must be `encodeURIComponent`-encoded.

Error handling in each queryFn:
```typescript
const res = await fetch(url, { headers: authHeaders() });
if (!res.ok) {
  const err = await res.json().catch(() => ({ error: res.statusText }));
  throw new Error(err.error ?? `Request failed: ${res.status}`);
}
return res.json();
```

Export all hooks and the `PAGE_SIZE` constant.

**1. Create src/routes/repo-browser/preview/DatasetPreview.tsx:**

This is the tab content wrapper. It manages:
- Validity check (gates entire preview)
- Split/config selection
- Search query state (debounced 300ms)
- Active filter state
- Column stats panel open/close state + selected column
- Page number state

Props:
```typescript
interface DatasetPreviewProps {
  repoId: string; // e.g. "user/my-dataset"
}
```

Logic flow:
1. Call `useDatasetIsValid(repoId)`. If `viewer === false`, render unavailable state.
2. Call `useDatasetSplits(repoId, isValid)`. Pre-select first split.
3. Maintain `searchQuery` state with 300ms debounce (use `useEffect` with `setTimeout`/`clearTimeout`). Track both `inputValue` (immediate) and `debouncedQuery` (delayed).
4. If `debouncedQuery` is non-empty, use `useDatasetSearch`; otherwise use `useDatasetRows`.
5. Pass data to `PreviewTable`.

**Preview unavailable state** (centered in tab content area):
```tsx
<div className="flex flex-col items-center justify-center h-full text-muted-foreground">
  <h2 className="text-xl font-semibold text-foreground mb-2">Preview unavailable</h2>
  <p className="text-sm">This dataset hasn't been processed for preview yet.</p>
</div>
```

**Toolbar row** (above table, single row, `flex items-center gap-3 mb-4`):
- Left: Search input (shadcn `Input`), width `w-72`, placeholder "Search rows...", `Search` icon from lucide-react inside left side. Include X clear button when search is active.
- Right: Split selector (shadcn `Select`) — only rendered if `splits.length > 1`. Label "Split:" inline. Value: `{config}/{split}`. On change, reset page to 0.

**Column filter state:** Maintain `activeFilters: Record<string, string>` — column name to filter value. When filters change, construct `where` clause for the `/filter` endpoint: `"column_name"='value'` joined by ` AND `. Use `useDatasetFilter` when filters are active (and no search query).

Priority: search > filter > plain rows. If search query is active, ignore filters. If filters are active but no search, use filter endpoint. Otherwise, use rows endpoint.
